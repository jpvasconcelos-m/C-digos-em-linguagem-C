//Código para dividir elementos em um array para pares e outro para ímpares.
#include <stdio.h>
#include<stdlib.h>
int main(){
int N,i,j;
int array[50],par[50],impar[50];
//Receber a quantidade de elementos:
scanf("%d",&N);
//Receber os elementos:
for(i=0;i<N;i++)
scanf("%d",&array[i]);
j=0;
//Transferir numeros para o array dos pares:
for(i=0;i<N;i++){
    if((array[i]%2)==0){
        par[j]=array[i];
        j++;
}}

//Loop para imprimir pares:
  for(i=0;i<j;i++)
  printf("pares[%d] = %d\n",i,par[i]);
  
j=0;
//Transferir numeros para o array dos pares:
for(i=0;i<N;i++){
    if((array[i]%2)!=0){
        impar[j]=array[i];
        j++;
}}  
//Loop para imprimir impares:
  for(i=0;i<j;i++)
  printf("impares[%d] = %d\n",i,impar[i]);
}
